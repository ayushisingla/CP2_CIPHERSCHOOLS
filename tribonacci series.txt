class Solution {
    public int tribonacci(int n) {
           
		if(n <= 1)  // if n <=1 just return n
            return n;
        
        int[] dp = new int[n+1]; //We alocate n+1 because we have T0 to Tn numbers
        
        dp[0] = 0;
        dp[1] = 1;
        dp[2] = 1;
        
        for(int i = 3; i <= n; ++i)
            dp[i] = dp[i-3] + dp[i-2] + dp[i-1]; 
        
        
        return dp[n]; 
    }
}
